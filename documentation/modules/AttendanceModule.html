import { Module } from '@nestjs/common';
import { AttendanceService } from './providers/attendance.service';

  imports: [
    TypeOrmModule.forFeature([Attendance]),
    StudentModule,
    LevelModule,
    UserTenantMembershipModule,
  ],
  providers: [AttendanceService, AttendanceResolver],
})
