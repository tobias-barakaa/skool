import { InputType, ObjectType, Field, Float, Int, ID } from '@nestjs/graphql';
import {
  IsOptional,
  IsString,
  IsInt,
  IsUUID,
  Min,
  Max,
  IsNumber,
  IsBoolean,
  IsDate,
  Length,
} from 'class-validator';

// ============================================================
// INPUT TYPE
// ============================================================

@InputType()
export class GetStudentMarksFilterDto {
  @Field({ nullable: true })
  @IsOptional()
  @IsString()
  @Length(4, 9) // e.g., "2024/2025"
  academicYear?: string;

  @Field(() => Int, { nullable: true })
  @IsOptional()
  @IsInt()
  @Min(1)
  @Max(3)
  term?: number;

  @Field({ nullable: true })
  @IsOptional()
  @IsUUID()
  tenantSubjectId?: string;

  @Field({ nullable: true })
  @IsOptional()
  @IsString()
  assessmentType?: string; // e.g., "CA", "EXAM"
}
