import { ObjectType, Field, ID, Float } from '@nestjs/graphql';
import { Entity, PrimaryGeneratedColumn, Column, ManyToOne, ManyToMany, JoinTable, JoinColumn, CreateDateColumn, UpdateDateColumn } from 'typeorm';
import { FeeStructure } from '../../fee-structure/entities/fee-structure.entity';
import { FeeBucket } from '../../fee-bucket/entities/fee-bucket.entity';
import { Term } from 'src/admin/academic_years/entities/terms.entity';

@Entity('fee_structure_items')
@ObjectType()
export class FeeStructureItem {
  @Field(() => ID)
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Field()
  @Column()
  tenantId: string;

  @Field()
  @Column()
  feeStructureId: string;

  @Field()
  @Column()
  feeBucketId: string;

  @Field(() => Float)
  @Column('decimal', { precision: 10, scale: 2 })
  amount: number;

  @Field()
  @Column({ default: true })
  isMandatory: boolean;

  @Field(() => FeeStructure)
  @ManyToOne(() => FeeStructure, structure => structure.items, { onDelete: 'CASCADE' })
  @JoinColumn({ name: 'feeStructureId' })
  feeStructure: FeeStructure;

  @Field(() => FeeBucket)
  @ManyToOne(() => FeeBucket)
  @JoinColumn({ name: 'feeBucketId' })
  feeBucket: FeeBucket;

  // CHANGED: Many-to-many relationship with terms
  @Field(() => [Term])
  @ManyToMany(() => Term)
  @JoinTable({
    name: 'fee_structure_item_terms',
    joinColumn: { name: 'fee_structure_item_id' },
    inverseJoinColumn: { name: 'term_id' }
  })
  terms: Term[];

  @Field(() => Date)
  @CreateDateColumn()
  createdAt: Date;

  @Field(() => Date)
  @UpdateDateColumn()
  updatedAt: Date;
}
